  <h2>Old Students List</h2>
  <table id="oldStudentsTable">
    <thead>
      <tr>
        <th>ID</th>
        <th>First Name</th>
        <th>Last Name</th>
        <th>Phone</th>
        <th>Email</th>
        <th>Program</th>
        <th>Start</th>
        <th>Renewal</th>
        <th>Actions</th>
      </tr>
    </thead>
    <tbody></tbody>
  </table>

  <script>
    async function loadOldStudents() {
      const res = await fetch('/api/old-students');
      const students = await res.json();
      const tbody = document.querySelector('#oldStudentsTable tbody');
      tbody.innerHTML = '';
      students.forEach(s => {
        const tr = document.createElement('tr');
        tr.innerHTML = `
          <td>${s.id}</td>
          <td>${s.first_name}</td>
          <td>${s.last_name}</td>
          <td>${s.phone || ''}</td>
          <td>${s.email || ''}</td>
          <td>${s.program || ''}</td>
          <td>${s.start_date || ''}</td>
          <td>${s.renewal_date || ''}</td>
          <td><button onclick="deleteOldStudent(${s.id})">Delete</button></td>
        `;
        tbody.appendChild(tr);
      });
    }

    async function deleteOldStudent(id) {
      if (!confirm('Delete this old student?')) return;
      await fetch('/api/old-students/' + id, { method: 'DELETE' });
      loadOldStudents();
    }

    loadOldStudents();
  </script>
