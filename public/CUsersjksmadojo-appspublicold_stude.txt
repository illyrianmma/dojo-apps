<script>
async function loadOld() {
  const res = await fetch('/api/old-students');
  const data = await res.json();
  const tbody = document.getElementById('tbody');
  tbody.innerHTML = '';
  data.forEach(s => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td>${s.id}</td>
      <td>${(s.first_name||'') + ' ' + (s.last_name||'')}</td>
      <td>${s.phone||''}</td>
      <td>${s.email||''}</td>
      <td>${s.program||''}</td>
      <td>${s.start_date||''}</td>
      <td>${s.renewal_date||''}</td>
      <td><button onclick="del(${s.id})">Delete</button></td>
    `;
    tbody.appendChild(tr);
  });
}
async function del(id){
  if (!confirm('Delete this old student?')) return;
  await fetch('/api/old-students/'+id, { method:'DELETE' });
  loadOld();
}
loadOld();
